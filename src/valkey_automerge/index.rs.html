11/// Prefix for index configuration keys
12const INDEX_CONFIG_PREFIX: &str = "am:index:config:";
13
14/// Prefix for shadow Hash keys
15const INDEX_KEY_PREFIX: &str = "am:idx:";
16
17/// Format for shadow index documents
18#[derive(Debug, Clone, Copy, PartialEq, Eq)]
19pub enum IndexFormat {
20    /// Store as Valkey Hash (flat key-value pairs)
21    Hash,
22    /// Store as JSON document (preserves structure)
23    Json,
24}
25
26impl IndexFormat {
27    fn from_str(s: &str) -> Option<Self> {
28        match s.to_lowercase().as_str() {
29            "hash" => Some(IndexFormat::Hash),
30            "json" => Some(IndexFormat::Json),
31            _ => None,
32        }
33    }
34
35    fn as_str(&self) -> &str {
36        match self {
37            IndexFormat::Hash => "hash",
38            IndexFormat::Json => "json",
39        }
40    }
41}

43/// Configuration for indexing a key pattern
44#[derive(Debug, Clone)]
45pub struct IndexConfig {
46    /// The key pattern (e.g., "article:*", "user:*")
47    pub pattern: String,
48    /// Whether indexing is enabled for this pattern
49    pub enabled: bool,
50    /// Paths to extract and index (e.g., ["title", "content", "author.name"])
51    pub paths: Vec<String>,
52    /// Format for shadow documents (hash or json)
53    pub format: IndexFormat,
54}
55
56impl IndexConfig {
57    /// Create a new index configuration (defaults to Hash format)
58    pub fn new(pattern: String, paths: Vec<String>) -> Self {
59        Self::new_with_format(pattern, paths, IndexFormat::Hash)
60    }
61
62    /// Create a new index configuration with specified format
63    pub fn new_with_format(pattern: String, paths: Vec<String>, format: IndexFormat) -> Self {
64        Self {
65            pattern,
66            enabled: true,
67            paths,
68            format,
69        }
70    }
71
72    /// Get the Redis key for storing this configuration
73    fn config_key(&self) -> String {
74        format!("{}{}", INDEX_CONFIG_PREFIX, self.pattern)
75    }
76
77    /// Save configuration to Redis
78    pub fn save(&self, ctx: &Context) -> ValkeyResult<()> {
79        let key = ctx.create_string(self.config_key());
80
81        // Store as Hash with fields: enabled, paths, format
82        ctx.call(
83            "HSET",
84            &[
85                &key,
86                &ctx.create_string("enabled"),
87                &ctx.create_string(if self.enabled { "1" } else { "0" }),
88            ],
89        )?;
90
91        let paths_str = self.paths.join(",");
92        ctx.call(
93            "HSET",
94            &[
95                &key,
96                &ctx.create_string("paths"),
97                &ctx.create_string(paths_str),
98            ],
99        )?;
100
101        ctx.call(
102            "HSET",
103            &[
104                &key,
105                &ctx.create_string("format"),
106                &ctx.create_string(self.format.as_str()),
107            ],
108        )?;
109
110        Ok(())
111    }
112
113    /// Load configuration from Redis
114    pub fn load(ctx: &Context, pattern: &str) -> ValkeyResult<Option<Self>> {
115        let key = ctx.create_string(format!("{}{}", INDEX_CONFIG_PREFIX, pattern));
116
117        // Check if key exists
118        let exists_result = ctx.call("EXISTS", &[&key])?;
119        let exists: i64 = match exists_result {
120            ValkeyValue::Integer(i) => i,
121            _ => return Err(ValkeyError::Str("Unexpected response from EXISTS")),
122        };
123
124        if exists == 0 {
125            return Ok(None);
126        }
127
128        // Get enabled field
129        let enabled_result = ctx.call("HGET", &[&key, &ctx.create_string("enabled")])?;
130        let enabled = match enabled_result {
131            ValkeyValue::SimpleString(s) | ValkeyValue::BulkString(s) => s == "1",
132            _ => true, // Default to enabled
133        };
134
135        // Get paths field
136        let paths_result = ctx.call("HGET", &[&key, &ctx.create_string("paths")])?;
137        let paths = match paths_result {
138            ValkeyValue::SimpleString(s) | ValkeyValue::BulkString(s) => s
139                .split(',')
140                .map(|p| p.to_string())
141                .filter(|p| !p.is_empty())
142                .collect(),
143            _ => Vec::new(),
144        };
145
146        // Get format field (default to Hash for backward compatibility)
147        let format_result = ctx.call("HGET", &[&key, &ctx.create_string("format")])?;
148        let format = match format_result {
149            ValkeyValue::SimpleString(s) | ValkeyValue::BulkString(s) => {
150                IndexFormat::from_str(&s).unwrap_or(IndexFormat::Hash)
151            }
152            _ => IndexFormat::Hash, // Default to Hash
153        };
154
155        Ok(Some(Self {
156            pattern: pattern.to_string(),
157            enabled,
158            paths,
159            format,
160        }))
161    }
162
163    /// Find the configuration that matches a given key
164    pub fn find_matching_config(ctx: &Context, key: &str) -> ValkeyResult<Option<Self>> {
165        // Get all configuration keys
166        let pattern = format!("{}*", INDEX_CONFIG_PREFIX);
167        let result = ctx.call("KEYS", &[&ctx.create_string(pattern)])?;
168
169        // Handle the Array result
170        let config_keys: Vec<ValkeyString> = match result {
171            ValkeyValue::Array(keys) => keys
172                .into_iter()
173                .filter_map(|v| match v {
174                    ValkeyValue::BulkString(s) => Some(ctx.create_string(s)),
175                    ValkeyValue::SimpleString(s) => Some(ctx.create_string(s)),
176                    _ => None,
177                })
178                .collect(),
179            _ => return Ok(None),
180        };
181
182        // Check each configuration to see if its pattern matches the key
183        for config_key in config_keys {
184            let config_key_str = config_key.to_string();
185            if let Some(pattern) = config_key_str.strip_prefix(INDEX_CONFIG_PREFIX) {
186                if Self::matches_pattern(key, pattern) {
187                    return Self::load(ctx, pattern);
188                }
189            }
190        }
191
192        Ok(None)
193    }
194
195    /// Check if a key matches a pattern (supports * wildcard)
196    fn matches_pattern(key: &str, pattern: &str) -> bool {
197        // Simple wildcard matching (* matches any characters)
198        if pattern == "*" {
199            return true;
200        }
201
202        if !pattern.contains('*') {
203            return key == pattern;
204        }
205
206        let parts: Vec<&str> = pattern.split('*').collect();
207        if parts.len() == 2 {
208            // Single wildcard: "prefix*" or "*suffix" or "prefix*suffix"
209            let prefix = parts[0];
210            let suffix = parts[1];
211
212            if prefix.is_empty() {
213                return key.ends_with(suffix);
214            } else if suffix.is_empty() {
215                return key.starts_with(prefix);
216            } else {
217                return key.starts_with(prefix) && key.ends_with(suffix);
218            }
219        }
220
221        // Multiple wildcards - simplified matching
222        let mut key_pos = 0;
223        for (i, part) in parts.iter().enumerate() {
224            if part.is_empty() {
225                continue;
226            }
227
228            if let Some(pos) = key[key_pos..].find(part) {
229                if i == 0 && pos != 0 {
230                    return false; // First part must match at start
231                }
232                key_pos += pos + part.len();
233            } else {
234                return false;
235            }
236        }
237
238        // Last part must match at end
239        if let Some(last) = parts.last() {
240            if !last.is_empty() && !key.ends_with(last) {
241                return false;
242            }
243        }
244
245        true
246    }
247}

249/// Extract configured paths from an Automerge document for Hash-based indexing
250pub fn extract_indexed_fields(
251    client: &RedisAutomergeClient,
252    paths: &[String],
253) -> HashMap<String, String> {
254    let mut fields = HashMap::new();
255
256    for path in paths {
257        // Try to get the value at this path
258        if let Ok(Some(value)) = client.get_text(path) {
259            // For nested paths, flatten with underscores for Hash field names
260            let field_name = path.replace('.', "_").replace('[', "_").replace(']', "");
261            fields.insert(field_name, value);
262        }
263        // Could also handle other types (int, bool, etc.) by converting to string
264        // For now, focus on text fields for full-text search
265    }
266
267    fields
268}

270/// Build a JSON document from configured paths for RedisJSON-based indexing
271///
272/// This extracts values from the Automerge document at the specified paths and
273/// builds a nested JSON object that preserves the path structure.
274///
275/// # Examples
276///
277/// Given paths `["title", "content", "meta.count", "tags"]`:
278/// ```json
279/// {
280///   "title": "Article Title",
281///   "content": "Article content...",
282///   "meta": {
283///     "count": 42
284///   },
285///   "tags": ["rust", "redis"]
286/// }
287/// ```
288pub fn build_json_document(
289    client: &RedisAutomergeClient,
290    paths: &[String],
291) -> Option<JsonValue> {
292    let mut root = Map::new();
293
294    for path in paths {
295        // Get typed value at this path
296        let typed_value = match client.get_typed_value(path) {
297            Ok(Some(val)) => val,
298            _ => continue, // Skip missing or error values
299        };
300
301        // Split path into segments
302        let segments: Vec<&str> = path.split('.').collect();
303
304        // Insert value at the correct nested location
305        insert_nested_value(&mut root, &segments, typed_value);
306    }
307
308    if root.is_empty() {
309        None
310    } else {
311        Some(JsonValue::Object(root))
312    }
313}
314
315/// Helper function to insert a typed value into a nested JSON object
316fn insert_nested_value(root: &mut Map<String, JsonValue>, segments: &[&str], value: TypedValue) {
317    if segments.is_empty() {
318        return;
319    }
320
321    if segments.len() == 1 {
322        // Base case: insert the value
323        root.insert(segments[0].to_string(), value.to_json());
324    } else {
325        // Recursive case: navigate or create nested objects
326        let key = segments[0].to_string();
327        let remaining = &segments[1..];
328
329        // Get or create the nested object
330        let nested = root
331            .entry(key.clone())
332            .or_insert_with(|| JsonValue::Object(Map::new()));
333
334        // Ensure it's an object
335        if let JsonValue::Object(nested_map) = nested {
336            insert_nested_value(nested_map, remaining, value);
337        } else {
338            // If there's a conflict (existing non-object value), replace it
339            let mut new_map = Map::new();
340            insert_nested_value(&mut new_map, remaining, value);
341            root.insert(key, JsonValue::Object(new_map));
342        }
343    }
344}

346/// Get the index key for a given Automerge key
347pub fn get_index_key(am_key: &str) -> String {
348    format!("{}{}", INDEX_KEY_PREFIX, am_key)
349}

351/// Update the JSON search index for a given Automerge key
352///
353/// This creates or updates a RedisJSON document with the configured fields.
354/// The JSON document preserves the nested structure of paths.
355///
356/// # Arguments
357///
358/// * `ctx` - Redis context for making commands
359/// * `am_key` - The Automerge document key
360/// * `client` - RedisAutomergeClient containing the document
361/// * `config` - Index configuration with paths to extract
362///
363/// # Returns
364///
365/// Returns `Ok(true)` if index was updated, `Ok(false)` if no fields were indexed
366pub fn update_json_index(
367    ctx: &Context,
368    am_key: &str,
369    client: &RedisAutomergeClient,
370    config: &IndexConfig,
371) -> ValkeyResult<bool> {
372    // Build JSON document from configured paths
373    let json_doc = match build_json_document(client, &config.paths) {
374        Some(doc) => doc,
375        None => {
376            // No fields to index - delete the index if it exists
377            let index_key = get_index_key(am_key);
378            ctx.call("DEL", &[&ctx.create_string(index_key)])?;
379            return Ok(false);
380        }
381    };
382
383    // Serialize JSON to string
384    let json_str = serde_json::to_string(&json_doc)
385        .map_err(|e| ValkeyError::String(format!("Failed to serialize JSON: {}", e)))?;
386
387    // Store as RedisJSON document
388    let index_key = get_index_key(am_key);
389    ctx.call(
390        "JSON.SET",
391        &[
392            &ctx.create_string(index_key),
393            &ctx.create_string("$"),
394            &ctx.create_string(json_str),
395        ],
396    )?;
397
398    Ok(true)
399}

401/// Update the search index for a given Automerge key
402///
403/// This is the main entry point for index updates. It dispatches to either
404/// Hash-based or JSON-based indexing depending on the configured format.
405pub fn update_search_index(
406    ctx: &Context,
407    am_key: &str,
408    client: &RedisAutomergeClient,
409) -> ValkeyResult<bool> {
410    // Find matching configuration
411    let config = match IndexConfig::find_matching_config(ctx, am_key)? {
412        Some(cfg) if cfg.enabled => cfg,
413        _ => return Ok(false), // No config or disabled
414    };
415
416    // Dispatch based on configured format
417    match config.format {
418        IndexFormat::Json => update_json_index(ctx, am_key, client, &config),
419        IndexFormat::Hash => update_hash_index(ctx, am_key, client, &config),
420    }
421}

423/// Update the Hash-based search index for a given Automerge key
424fn update_hash_index(
425    ctx: &Context,
426    am_key: &str,
427    client: &RedisAutomergeClient,
428    config: &IndexConfig,
429) -> ValkeyResult<bool> {
430    // Extract configured fields
431    let fields = extract_indexed_fields(client, &config.paths);
432
433    if fields.is_empty() {
434        // No fields to index - delete the index Hash
435        let index_key = get_index_key(am_key);
436        ctx.call("DEL", &[&ctx.create_string(index_key)])?;
437        return Ok(false);
438    }
439
440    // Update Hash with extracted fields
441    let index_key = get_index_key(am_key);
442    let index_key_rs = ctx.create_string(index_key.clone());
443
444    // Delete existing Hash first to ensure clean state
445    ctx.call("DEL", &[&index_key_rs])?;
446
447    // Set each field
448    for (field, value) in &fields {
449        ctx.call(
450            "HSET",
451            &[
452                &index_key_rs,
453                &ctx.create_string(field.clone()),
454                &ctx.create_string(value.clone()),
455            ],
456        )?;
457    }
458
459    Ok(true)
460}

462/// Delete the search index Hash for a given Automerge key
463pub fn delete_search_index(ctx: &Context, am_key: &str) -> ValkeyResult<()> {
464    let index_key = get_index_key(am_key);
465    ctx.call("DEL", &[&ctx.create_string(index_key)])?;
466    Ok(())
